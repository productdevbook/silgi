## Setup

```ts
import { createQueryUtils } from "katman/tanstack-query"

const utils = createQueryUtils(client)
```

## Queries

```tsx
import { useQuery } from "@tanstack/react-query"

function UserList() {
  const { data } = useQuery(
    utils.users.list.queryOptions({ input: { limit: 10 } }) // [!code highlight]
  )

  return <ul>{data?.map(u => <li key={u.id}>{u.name}</li>)}</ul>
}
```

## Mutations

```tsx
import { useMutation, useQueryClient } from "@tanstack/react-query"

function CreateUser() {
  const queryClient = useQueryClient()

  const mutation = useMutation(
    utils.users.create.mutationOptions({
      onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: utils.users.key() }) // [!code highlight]
      },
    })
  )

  return <button onClick={() => mutation.mutate({ name: "Alice" })}>Create</button>
}
```

## Prefetching

```ts
await queryClient.prefetchQuery(
  utils.users.list.queryOptions({ input: { limit: 10 } })
)
```

## Cache Invalidation

```ts
// Invalidate all user queries
queryClient.invalidateQueries({ queryKey: utils.users.key() })

// Invalidate a specific query
queryClient.invalidateQueries({
  queryKey: utils.users.list.queryKey({ limit: 10 }),
})
```

<Callout>
Works with `@tanstack/react-query`, `@tanstack/vue-query`, `@tanstack/solid-query`, and `@tanstack/svelte-query`. The API is the same.
</Callout>
