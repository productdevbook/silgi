## Enable in serve()

```ts
k.serve(appRouter, { port: 3000, ws: true })
```

HTTP and WebSocket run on the same port.

## Protocol

```
Client → Server: { id: "1", path: "users/list", input: { limit: 10 } }
Server → Client: { id: "1", result: [{ id: 1, name: "Alice" }] }

// Streaming (subscription)
Server → Client: { id: "2", data: { count: 1 } }
Server → Client: { id: "2", data: { count: 2 } }
Server → Client: { id: "2", data: null, done: true }

// Error
Server → Client: { id: "3", error: { code: "NOT_FOUND", status: 404 } }
```

## Client Usage

```ts
const ws = new WebSocket("ws://localhost:3000")

ws.onopen = () => {
  ws.send(JSON.stringify({
    id: "1",
    path: "users/list",
    input: { limit: 10 },
  }))
}

ws.onmessage = (event) => {
  const { id, result, error, data, done } = JSON.parse(event.data)
  if (error) console.error(error)
  else if (data !== undefined) console.log("Stream:", data)
  else console.log("Result:", result)
}
```

## Standalone Attachment

Attach to any existing Node.js HTTP server:

```ts
import { createServer } from "node:http"
import { attachWebSocket } from "katman/ws"

const server = createServer(httpHandler)
attachWebSocket(server, appRouter, { binary: true }) // optional msgpack
server.listen(3000)
```
