## Setup

### 1. Silgi instance

```ts title="server/rpc/instance.ts"
import { silgi } from 'silgi'
import { todos } from './todos/schema'

export const s = silgi({
  context: () => ({ todos }),
})
```

### 2. Procedures with tracing

```ts twoslash title="server/rpc/todos/index.ts"
// @noErrors
import { trace } from 'silgi/analytics'
import { s } from '../instance'
import { CreateTodoSchema, TodoIdSchema, TodoListSchema, TodoSchema, OkSchema, getNextId } from './schema'

export const list = s
  .$route({ method: 'GET' })
  .$output(TodoListSchema)
  .$resolve(async ({ ctx }) => {
    return trace(ctx, 'db.todos.findMany', () => [...ctx.todos])
  })

export const create = s
  .$input(CreateTodoSchema)
  .$output(TodoSchema)
  .$resolve(async ({ input, ctx }) => {
    const todo = { id: getNextId(), title: input.title, completed: false, createdAt: new Date().toISOString() }
    await trace(ctx, 'db.todos.insert', () => ctx.todos.push(todo))
    return todo
  })

export const toggle = s
  .$input(TodoIdSchema)
  .$output(TodoSchema)
  .$errors({ NOT_FOUND: 404 })
  .$resolve(async ({ input, ctx, fail }) => {
    const todo = await trace(ctx, 'db.todos.findById', () => ctx.todos.find((t) => t.id === input.id))
    if (!todo) return fail('NOT_FOUND')
    todo.completed = !todo.completed
    return todo
  })

export const remove = s
  .$input(TodoIdSchema)
  .$output(OkSchema)
  .$errors({ NOT_FOUND: 404 })
  .$resolve(async ({ input, ctx, fail }) => {
    const idx = ctx.todos.findIndex((t) => t.id === input.id)
    if (idx === -1) return fail('NOT_FOUND')
    ctx.todos.splice(idx, 1)
    return { ok: true }
  })
```

### 3. Router

```ts title="server/rpc/router.ts"
import * as auth from './auth'
import { s } from './instance'
import * as todos from './todos'

export const appRouter = s.router({
  auth: {
    login: auth.doLogin,
    logout: auth.doLogout,
    me: auth.me,
  },
  todos: {
    list: todos.list,
    create: todos.create,
    toggle: todos.toggle,
    remove: todos.remove,
  },
})

export type AppRouter = typeof appRouter
```

### 4. Server entry

```ts title="server.ts"
import { s } from './server/rpc/instance'
import { appRouter } from './server/rpc/router'

export default {
  fetch: s.handler(appRouter, {
    scalar: true,
    analytics: {
      auth: process.env.ANALYTICS_TOKEN,
    },
  }),
}
```

### 5. Nitro config

```ts title="nitro.config.ts"
import { defineNitroConfig } from 'nitro/config'

export default defineNitroConfig({
  compatibilityDate: '2025-01-01',
})
```

<Callout>Nitro detects the `fetch` method on the default export and routes all requests through it. The analytics dashboard is available at `/analytics`.</Callout>

## Coexisting with existing routes

When you add Silgi to an app that already has its own routes — GraphQL endpoints, REST API routes, file-system routes — the default `fetch` export takes over **all** requests. To let unmatched routes fall through to Nitro's normal routing:

```ts twoslash title="server.ts"
// @noErrors
import { silgi } from 'silgi'

const s = silgi()
const appRouter = s.router({
  health: s.$resolve(() => ({ status: 'ok' })),
})

const silgiHandler = s.handler(appRouter)

export default {
  fetch: async (request: Request) => {
    const response = await silgiHandler(request)
    if (response.status === 404) return
    return response
  },
}
```
