## Step by Step

<Steps>
<Step>

### Create the katman instance

Define your context factory — this runs on every request and populates `ctx`.

```ts title="src/rpc.ts"
import { katman } from "katman"

export const k = katman({
  context: (req) => ({
    headers: Object.fromEntries(req.headers) as Record<string, string>,
  }),
})

export const { query, mutation, guard, wrap, router } = k
```

</Step>
<Step>

### Define procedures

Each endpoint is a **procedure** — a `query` (read) or `mutation` (write).

```ts title="src/routes/users.ts"
import { z } from "zod"
import { query, mutation, guard } from "../rpc"
import { KatmanError } from "katman"

export const auth = guard(async (ctx) => {
  const token = ctx.headers.authorization?.replace("Bearer ", "")
  if (!token) throw new KatmanError("UNAUTHORIZED")
  return { userId: 1 } // [!code highlight]
})

export const listUsers = query(
  z.object({ limit: z.number().min(1).max(100).optional() }),
  ({ input }) => db.users.findMany({ take: input.limit ?? 10 }),
)

export const createUser = mutation({
  use: [auth], // [!code highlight]
  input: z.object({
    name: z.string().min(1),
    email: z.string().email(),
  }),
  errors: { CONFLICT: 409 },
  resolve: ({ input, fail }) => {
    if (db.users.exists(input.email)) fail("CONFLICT") // [!code highlight]
    return db.users.create(input)
  },
})
```

</Step>
<Step>

### Build the router

Nest procedures into a tree. The structure becomes the API path.

```ts title="src/router.ts"
import { router } from "./rpc"
import { listUsers, createUser } from "./routes/users"

export const appRouter = router({
  users: {
    list: listUsers,   // → POST /users/list
    create: createUser, // → POST /users/create
  },
})

export type AppRouter = typeof appRouter
```

</Step>
<Step>

### Start the server

```ts title="src/server.ts"
import { k } from "./rpc"
import { appRouter } from "./router"

k.serve(appRouter, {
  port: 3000,
  scalar: true,  // → /reference (API docs)
  ws: true,      // → WebSocket RPC on same port
})
```

```
Katman server running at http://127.0.0.1:3000
Scalar API Reference at http://127.0.0.1:3000/reference
```

</Step>
<Step>

### Create the client

```ts title="src/client.ts"
import { createClient } from "katman/client"
import { createLink } from "katman/client/ofetch"
import type { InferClient } from "katman"
import type { AppRouter } from "./router"

const link = createLink({ url: "http://localhost:3000" })
export const client = createClient<InferClient<AppRouter>>(link) // [!code highlight]

// Fully typed — autocomplete works!
const users = await client.users.list({ limit: 5 })
const user = await client.users.create({ name: "Alice", email: "a@test.com" })
```

</Step>
</Steps>
