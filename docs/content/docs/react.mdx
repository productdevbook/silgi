## createAction

Wrap a single procedure as a server action. Returns `[error, data]` tuples.

```ts title="app/actions.ts"
"use server"
import { createAction } from "katman/react"
import { appRouter } from "./router"

export const listUsers = createAction(appRouter.users.list)
export const createUser = createAction(appRouter.users.create)
```

```tsx title="app/page.tsx"
import { listUsers } from "./actions"

export default async function Page() {
  const [error, users] = await listUsers({ limit: 10 }) // [!code highlight]
  if (error) return <p>Error: {error.message}</p>

  return (
    <ul>
      {users.map(u => <li key={u.id}>{u.name}</li>)}
    </ul>
  )
}
```

## createActions

Convert an entire router at once:

```ts
import { createActions } from "katman/react"

const actions = createActions(appRouter)

const [err, users] = await actions.users.list({ limit: 10 })
const [err2, user] = await actions.users.create({ name: "Alice" })
```

## createFormAction

For `<form action={...}>`:

```ts title="app/actions.ts"
"use server"
import { createFormAction } from "katman/react"

export const submitUser = createFormAction(appRouter.users.create)
```

```tsx
<form action={submitUser}>
  <input name="name" />
  <input name="email" type="email" />
  <button type="submit">Create</button>
</form>
```

<Callout>
Supports bracket notation in form fields: `user[name]` becomes `{ user: { name: value } }`.
Next.js redirect/notFound and TanStack Router errors are automatically rethrown.
</Callout>
