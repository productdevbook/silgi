<Callout type="info" title="Why Katman?">
Katman compiles your middleware into a pre-linked chain at startup. No per-request closures. No runtime overhead. Just raw speed.
</Callout>

```ts title="server.ts"
import { katman, KatmanError } from "katman"
import { z } from "zod"

const k = katman({
  context: (req) => ({ db: getDB() }),
})

const { query, mutation, guard, router } = k

const auth = guard(async (ctx) => {
  const user = await verify(ctx.headers.authorization)
  if (!user) throw new KatmanError("UNAUTHORIZED")
  return { user } // [!code highlight]
})

const listUsers = query(
  z.object({ limit: z.number().optional() }),
  ({ input, ctx }) => ctx.db.users.findMany({ take: input.limit }),
)

const createUser = mutation({
  use: [auth],
  input: z.object({ name: z.string() }),
  errors: { CONFLICT: 409 },
  resolve: ({ input, ctx, fail }) => {
    if (ctx.db.users.exists(input.name)) fail("CONFLICT") // [!code highlight]
    return ctx.db.users.create(input)
  },
})

k.serve(router({ users: { list: listUsers, create: createUser } }), {
  port: 3000,
  scalar: true, // API docs at /reference
  ws: true,     // WebSocket on same port
})
```
