---
title: Message Port
description: Use Silgi over MessagePort — Electron, browser extensions, Web Workers.
---

Silgi can run over any `MessagePort`-compatible channel — Electron IPC, browser extension ports, Web Workers, iframes.

## Server side

Attach your router to a port:

```ts twoslash
// @noErrors
import { createHandler } from 'silgi/message-port'
import { appRouter } from './router'

const port = getPort() // MessagePort, Worker, chrome.runtime.Port, etc.

createHandler(appRouter, port, {
  context: () => ({
    db: getDB(),
  }),
})
```

## Client side

Connect to the same port from the other end:

```ts twoslash
// @noErrors
import { MessagePortLink } from 'silgi/message-port'
import { createClient } from 'silgi/client'
import type { AppRouter } from './router'

const port = getPort()

const client = createClient<AppRouter>({
  links: [new MessagePortLink(port)],
})

const users = await client.users.list()
```

## Electron example

```ts twoslash
// @noErrors
// main.ts (main process)
import { createHandler } from 'silgi/message-port'

ipcMain.handle('rpc-port', (event) => {
  const { port1, port2 } = new MessageChannelMain()
  createHandler(appRouter, port1, { context: () => ({}) })
  event.senderFrame.postMessage('rpc-port', null, [port2])
})
```

```ts twoslash
// @noErrors
// renderer.ts (renderer process)
import { MessagePortLink } from 'silgi/message-port'
import { createClient } from 'silgi/client'

const port = await getElectronPort()
const client = createClient<AppRouter>({
  links: [new MessagePortLink(port)],
})
```
